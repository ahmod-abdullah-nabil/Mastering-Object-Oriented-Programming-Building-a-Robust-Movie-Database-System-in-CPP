    WITHOUT Encapsulation (BAD):
    ??????????????????????????????
    
    Movie movie;
    movie.rating = 100;  // ALLOWED - Can set invalid value!
    movie.name = "";     // ALLOWED - Can set empty name!

    ??????????????????????????????
    
    Movie movie;
    movie.rating = 100;  // ? COMPILER ERROR - rating is private!
    
    movie.setRating(100); // ? Allowed, but validation happens:
                          //   if (rating > 5) rating = 5;

    int r = movie.getRating(); // ? Controlled read access
